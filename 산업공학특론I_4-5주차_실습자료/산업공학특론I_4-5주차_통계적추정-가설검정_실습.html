1-1. MLE

# 지수분포 난수 생성
x <- rexp(10000,rate = 0.1)
# 로그우도함수 정의
mle = function(param){
  -sum(dexp(x,rate = param, log = TRUE)) # pdf의 density
}
# 최적화 수행하여 모수 추정
# ? optimize

mle_opt = optimize(mle,c(0,10))
mle_opt # minimum -> 추정량 : Lambda
## $minimum
## [1] 0.09902554
## 
## $objective
## [1] 33125.02
# objective : Likelihood Function Value
#optim()

1-2. LSE

# 예측값 기반 최소제곱함수 정의
lse <- function(param){
  pred <- param*exp(-param*x)  # 지수분포 Term
  sum((x-pred)^2)
}
lse_opt <- optimize(lse,c(0,10))
lse_opt # Minimum을 이용했을 때의 Likelihood 값
## $minimum
## [1] 0.097475
## 
## $objective
## [1] 2027564
# 최적화 수행하여 모수 추정
